**STEP 1: Comprehensive spot-check — 15 specific numbers across all 3 files.**

```bash
python3 << 'SPOTCHECK'
import json, subprocess, sys

gpt = json.load(open('results/analysis/paper_data_gpt41mini.json'))
cmp = json.load(open('results/analysis/cross_model_comparison.json'))
qwen = json.load(open('results/analysis/paper_data.json'))

pc = gpt['primary_campaign']
ov = pc['overall']

# Build check list: (label, file, pattern_present, pattern_absent)
checks = []

rate_str = f"{ov['pass_rate']*100:.1f}"
total_str = str(ov['total'])
pass_str = str(ov['pass'])
chi2_str = f"{cmp['overall']['chi_squared']:.2f}"
h_str = f"{cmp['overall']['cohens_h']:.2f}"
bf_count = str(ov['by_status'].get('BUILD_FAIL', ov.get('build_fail', 0)))
vf_count = str(ov['by_status'].get('VERIFY_FAIL', ov.get('verify_fail', 0)))

# Aggregate
combined_total = 710 + ov['total']
combined_pass = 272 + ov['pass']

# Per-kernel
pkm = cmp['per_kernel_matrix']
both_pass = str(pkm['counts']['both_pass'])
common_kernels = str(pkm['total_common_kernels'])

for fname in ['docs/paper/latex/overleaf.tex', 'docs/paper/latex/paper.tex']:
    checks.append((f'{fname}: new GPT rate {rate_str}%', fname, rate_str, None))
    checks.append((f'{fname}: new GPT total {total_str}', fname, total_str, None))
    checks.append((f'{fname}: new chi2 {chi2_str}', fname, chi2_str, None))
    checks.append((f'{fname}: no old rate 29.2 in GPT context', fname, None, '29.2'))
    checks.append((f'{fname}: both_pass {both_pass}', fname, f'{both_pass} are solved', None))

checks.append(('appendices.tex: no old chi2 10.97', 'docs/paper/latex/appendices.tex', None, '10.97'))

passed = 0
failed = 0
for label, fname, present, absent in checks:
    ok = True
    if present:
        r = subprocess.run(['grep', '-c', present, fname], capture_output=True, text=True)
        count = int(r.stdout.strip() or '0')
        if count == 0:
            print(f'FAIL: {label} — pattern "{present}" not found')
            ok = False
    if absent:
        r = subprocess.run(['grep', '-c', absent, fname], capture_output=True, text=True)
        count = int(r.stdout.strip() or '0')
        # Filter: only flag if it appears in a GPT context (rough heuristic)
        if count > 0 and absent == '29.2':
            r2 = subprocess.run(['grep', '-n', absent, fname], capture_output=True, text=True)
            gpt_lines = [l for l in r2.stdout.split('\n') if 'GPT' in l or 'gpt' in l]
            if gpt_lines:
                print(f'FAIL: {label} — "{absent}" still appears in GPT context ({len(gpt_lines)} lines)')
                ok = False
        elif count > 0 and absent != '29.2':
            print(f'FAIL: {label} — "{absent}" still appears {count} times')
            ok = False
    if ok:
        passed += 1
    else:
        failed += 1

print(f'\n=== SPOT CHECK: {passed} passed, {failed} failed ===')
if failed > 0:
    print('FIX FAILURES BEFORE COMMITTING')
    sys.exit(1)
SPOTCHECK
```
