   def _wilson_ci(successes: int, total: int, z: float = 1.96) -> tuple[float, float]:
       if total == 0:
           return (0.0, 0.0)
       p_hat = successes / total
       denom = 1 + z**2 / total
       center = (p_hat + z**2 / (2 * total)) / denom
       spread = z * ((p_hat * (1 - p_hat) / total + z**2 / (4 * total**2)) ** 0.5) / denom
